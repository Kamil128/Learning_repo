Rozkład Bernulliego¶

Rozkad zero-jedynkowy dyskretny rozkad prawdopodobiestwa, szczeglny przypadek rozkadu dwupunktowego, dla ktrego zmienna losowa przyjmuje tylko wartoci: 0 i 1. Jest on na przykad rezultatem dowiadczenia (zwanego prb Bernoulliego), w wyniku ktrego okrelone zdarzenie

W krajach anglojzycznych rozkad ten nazywany jest Bernoulli distribution. W polskim pimiennictwie jednak zwyczajowo rozkad Bernoulliego oznacza rozkad dwumianowy.

import numpy as np
from scipy.stats import bernoulli
import matplotlib.pyplot as plt
fig, (ax1, ax2, ax3) = plt.subplots(1, 3, figsize=(15, 4))

# Parametr
p = 0.7

# Generowanie prbki
proba = bernoulli.rvs(p, size=1_000)  # Random variates.

ax1.set_title("Histogram losowej prbki")
ax1.hist(proba, weights=np.ones(len(proba)) / len(proba))

# Generowanie statystyk
mean, var, skew, kurt = bernoulli.stats(p, moments='mvsk')

print(f"Podstawowe statystyki: \nrednia {mean}, \nwariancja: {var}, \nskono: {skew}, \nkurtoza: {kurt}")

# CDF - Cumulative distribution function. - dystrybuanta
# Percent point function (inverse of cdf  percentiles).
x = np.arange(bernoulli.ppf(0., p),
                bernoulli.ppf(1, p)+1)
# x = np.array([0, 1])

# pmf - probability mass function
# Funkcja masy prawdopodobiestwa (ang. probability mass function, pmf)  funkcja dajca dla kadej liczby rzeczywistej 
# u prawdopodobiestwo, e dana dyskretna zmienna losowa przyjmie warto 

ax2.set_title("Funkcja masy prawdopodobiestwa")
ax2.scatter(x, bernoulli.pmf(x, p), label='funkcja masy prawdopodobiestwa', s=50.0)
# ax.vlines(x, 0, bernoulli.pmf(x, p), colors='b', lw=5, alpha=0.5)

# rv = bernoulli(p)
# ax2.vlines(x, 0, rv.pmf(x), colors='k', linestyles='-', lw=1, label='frozen pmf')

# ax.hist(proba, bins=50, color='red', alpha=0.1) 

# ax2.legend(loc='best', frameon=False)

xd = np.linspace(-1, 2, 100)

ax3.set_title("Dystrybuanta")
ax3.plot(xd, bernoulli.cdf(xd, p=p))

x_lim = [-1, 2]
ax1.set_xlim(x_lim)
ax2.set_xlim(x_lim)
ax3.set_xlim(x_lim)

y_lim = [0, 1.1]
ax1.set_ylim(y_lim)
ax2.set_ylim(y_lim)
ax3.set_ylim(y_lim)

import numpy as np
from scipy.stats import binom
import matplotlib.pyplot as plt
fig, (ax1, ax2, ax3) = plt.subplots(1, 3, figsize=(15, 4))

# parametry rozkdu
n = 10  # ilo prb
p = 0.3  # prawdopodobiestwo sukcesu w pojedyczej prbie

# Generowanie prbki
proba = binom.rvs(n=n, p=p, size=10_000)  # Random variates.

ax1.set_title("Histogram losowej prbki")
ax1.hist(proba, weights=np.ones(len(proba)) / len(proba), bins=50)

# Generowanie teoretycznych statystyk
mean, var, skew, kurt = binom.stats(p=p, n=n, moments='mvsk')

print(f"Podstawowe statystyki: \nrednia {mean}, \nwariancja: {var}, \nskono: {skew}, \nkurtoza: {kurt}")

# Funkcja masy prawdopodobiestwa
p_min = binom.ppf(0., n, p)
p_max = binom.ppf(1, n, p)+2
x = np.arange(p_min, p_max)
ax2.set_title("Funkcja masy prawdopodobiestwa")
ax2.scatter(x, binom.pmf(x, p=p, n=n), label='funkcja masy prawdopodobiestwa', s=50.0)

# Dystrybuanta
xd = np.linspace(p_min, p_max, 100)
ax3.set_title("Dystrybuanta")
ax3.plot(xd, binom.cdf(xd, p=p, n=n))

x_lim = [p_min, p_max]
ax1.set_xlim(x_lim)
ax2.set_xlim(x_lim)
ax3.set_xlim(x_lim)

y_lim = [0, 1.1]
ax1.set_ylim(y_lim)
ax2.set_ylim(y_lim)
ax3.set_ylim(y_lim)

plt.show()

rednia 3.0, 
wariancja: 2.0999999999999996, 
skono: 0.2760262237369417,

Rozkład Poissona¶

Rozkad Poissona jest rozkadem dyskretnym (skokowym), ktry wyraa prawdopodobiestwo zdarze nastpujcych po sobie z dan czstotliwoci (ilo zdarze na jednostk czasow) w danym czasie. Zdarzenia zachodz niezalenie, tzn. e czas nastpnego zdarzenia nie zaley od tego kiedy wystpio poprzednie zdarzenie.

Rozkad Poissona modeluje liczb zdarze, ktre wystpuj w cigu jednostkowego odcinka czasu przy zaoeniu, e

  • liczby zdarze wystpujcych na rozcznych podprzedziaach s niezalene,
  • prawdopodobiestwa wystpienia zdarzenia na krtkim podprzedziale o dugoci h szacuje si przez h + o(h),
  • prawdopodobiestwo wystpienia wicej ni jednego zdarzenia na krtkim podprzedziale o dugoci h szacuje. si przez o(h).

Rozwamy taki przykad: obserwujemy pewien przystanek autobusowy i stwierdzamy, e rednio autobusy przyjedaj z czstotliwoci =4 autobusy na godzin. Zdarza si jednak, e autobus nie przyjedzie- ulegnie wypadkowi lub si spni. Wtedy moe okaza si, e w jedn godzin przyjad 3 autobusy, a w nastpn 5. Jednak rednio w kad godzin bdziemy obserwowa 4 autobusy na tym przystanku. Zmienna opisujca kiedy przyjedzie autobus jest wanie zmienn z rozkadu Poissona o intensywnoci =4.

W pewnym hipermarkecie zatrudnionych jest 90 pracownikw. Prawdopodobiestwo e pracownik si spni danego dnia wynosi p=0.03. Jakie jest prawdopodobiestwo,e w losowo wybranym miesicu do pracy spnio si: 2 pracownikw

import numpy as np
from scipy.stats import poisson
import matplotlib.pyplot as plt
fig, (ax1, ax2, ax3) = plt.subplots(1, 3, figsize=(15, 4))

# parametry rozkadu
mu = 0.03*90  # rednia

# Generowanie prbki
proba = poisson.rvs(mu=mu, size=10_000)  # Random sample.

ax1.set_title("Histogram losowej prbki")
ax1.hist(proba, weights=np.ones(len(proba)) / len(proba), bins=50)

# Generowanie teoretycznych statystyk
mean, var, skew, kurt = poisson.stats(mu=mu, moments='mvsk')

print(f"Podstawowe statystyki: \nrednia {mean}, \nwariancja: {var}, \nskono: {skew}, \nkurtoza: {kurt}")

# Funkcja masy prawdopodobiestwa
p_min = poisson.ppf(0., mu)
p_max = poisson.ppf(0.99, mu)+2
x = np.arange(p_min, p_max)
ax2.set_title("Funkcja masy prawdopodobiestwa")
ax2.scatter(x, poisson.pmf(x, mu), label='funkcja masy prawdopodobiestwa', s=50.0)

# Dystrybuanta
xd = np.linspace(p_min, p_max, 100)
ax3.set_title("Dystrybuanta")
ax3.plot(xd, poisson.cdf(xd, mu=mu))

x_lim = [p_min, p_max]
ax1.set_xlim(x_lim)
ax2.set_xlim(x_lim)
ax3.set_xlim(x_lim)

y_lim = [0, 1.1]
ax1.set_ylim(y_lim)
ax2.set_ylim(y_lim)
ax3.set_ylim(y_lim)

from scipy.stats import t
import matplotlib.pyplot as plt
import numpy as np

fig, (ax1, ax3) = plt.subplots(1, 2, figsize=(10, 4))

# parametry rozkadu
df=2.5
# Generowanie prbki
proba = t.rvs(df=df, size=10_000)  # Random sample.

ax1.set_title("Histogram losowej prbki")
ax1.hist(proba, bins=50, density=True, histtype='stepfilled', alpha=0.2)

# Generowanie teoretycznych statystyk
mean, var, skew, kurt = t.stats(df=df, moments='mvsk')

print(f"Podstawowe statystyki: \nrednia {mean}, \nwariancja: {var}, \nskono: {skew}, \nkurtoza: {kurt}")

# Funkcja prawdopodobiestwa
# pdf - Probability density function.
# 
p_min = t.ppf(0.01, df)
p_max = t.ppf(0.99, df)
x = np.linspace(p_min, p_max, 100)
ax1.set_title("Funkcja masy prawdopodobiestwa")
ax1.plot(x, t.pdf(x, df), label='funkcja prawdopodobiestwa')

# Dystrybuanta
xd = np.linspace(p_min, p_max, 100)
ax3.set_title("Dystrybuanta")
ax3.plot(xd, t.cdf(xd, df=df))

x_lim = [p_min, p_max]
ax1.set_xlim(x_lim)
ax3.set_xlim(x_lim)

ax1.set_ylim([0, 0.6])
ax3.set_ylim([0, 1.1])

plt.show()

Rozkład normalny¶

Rozkad normalny, inaczej zwany rozkadem Gaussa, krzyw Gaussa jest najwaniejszym rozkadem teoretycznym prawdopodobiestwa w statystyce. Rozkad normalny jest te najbardziej intuicyjnym rozkadem statystycznym. W wielkim skrcie opisuje on sytuacje w wiecie, gdzie wikszo przypadkw jest bliska redniemu wynikowi, a im dany wynik bardziej odchyla si od redniej tym jest mniej reprezentowany. Najwicej jest przypadkw blisko przecitnej. Im dalej oddalamy si od redniego wyniku, tym przypadkw jest mniej. Mona to z atwoci odnie do rzeczywistych sytuacji.
